                </div>
                <button onclick="setupSeats()">🪑 設定座位</button>
            </div>

            <div class="zoom-controls">
                <label>縮放：</label>
                <input type="range" id="zoomSlider" min="50" max="150" value="100" oninput="adjustZoom(this.value)">
                <span id="zoomVal">100%</span>
                <button onclick="shuffleSeats()">🔀 隨機換位</button>

    <script>
        // --- 全域變數 ---
        let students = JSON.parse(localStorage.getItem('students')) || [];
        let studentPoints = {}; // 模擬格數資料
        let showPoints = true; 
        let rows = 6, cols = 6;

        // --- 初始化 ---
        window.onload = () => {
            document.getElementById('selectedDate').valueAsDate = new Date();
            // 模擬從 Google Sheets 抓到的資料
            fetchMockPoints();
            renderSeats();
        };

        // --- 核心功能：切換格數顯示 ---
        function togglePointsDisplay() {
            showPoints = document.getElementById('togglePoints').checked;
            console.log("格數顯示狀態:", showPoints);
            renderSeats(); // 立即重新渲染
        }

        // --- 渲染座位表 (修正後) ---
        function renderSeats() {
            const grid = document.getElementById('seatGrid');
            grid.style.gridTemplateColumns = `repeat(${cols}, 120px)`;
            grid.innerHTML = '';

            for (let r = 1; r <= rows; r++) {
                for (let c = 1; c <= cols; c++) {
                    const cell = document.createElement('div');
                    cell.className = 'seat-cell';
                    
                    const student = students.find(s => s.row === r && s.col === c);
                    if (student) {
                        // 判斷是否要顯示格數 HTML
                        const pointsHtml = showPoints ? 
                            `<div class="seat-points">格數: ${studentPoints[student.number] || 0}</div>` : '';

                        cell.innerHTML = `
                            <div class="student-seat-card">
                                <div class="seat-number">No.${student.number}</div>
                                <div class="seat-name">${student.name}</div>
                                ${pointsHtml}
                            </div>
                        `;
                    } else {
                        cell.innerHTML = '<span style="color:#ccc; font-size:12px">空位</span>';
                    }
                    grid.appendChild(cell);
                }
            }
        }

        // --- 其他輔助功能 ---
        function addStudent() {
            const name = document.getElementById('studentName').value;
            const number = document.getElementById('studentNumber').value;
            if(!name || !number) return alert('請輸入姓名座號');
            
            // 找空位放
            students.push({ name, number, row: 1, col: 1 });
            saveAndRender();
        }

        function setupSeats() {
            rows = parseInt(document.getElementById('rows').value);
            cols = parseInt(document.getElementById('cols').value);
            renderSeats();
        }

        function adjustZoom(val) {
            document.getElementById('zoomVal').innerText = val + '%';
            document.getElementById('seatGrid').style.transform = `scale(${val/100})`;
        }

        function shuffleSeats() {
            let seats = [];
            for(let r=1; r<=rows; r++) for(let c=1; c<=cols; c++) seats.push({r, c});
            
            students.forEach(s => {
                if(seats.length > 0) {
                    let idx = Math.floor(Math.random() * seats.length);
                    let pos = seats.splice(idx, 1)[0];
                    s.row = pos.r;
                    s.col = pos.c;
                }
            });
            saveAndRender();
        }

        function saveAndRender() {
            localStorage.setItem('students', JSON.stringify(students));
            renderSeats();
        }

        function fetchMockPoints() {
            // 這裡模擬從 Google Sheets 讀取的資料
            for(let i=1; i<=50; i++) {
                studentPoints[i] = Math.floor(Math.random() * 20);
            }
        }
